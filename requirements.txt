# Step 1 — Install CUDA-enabled PyTorch (do NOT use requirements.txt for this):
#   pip uninstall torch torchvision -y
#   pip cache purge
#   pip install torch==2.11.0 torchvision==0.26.0 --index-url https://download.pytorch.org/whl/cu130
#
# Step 2 — Install CUDA-enabled PaddlePaddle (do NOT use requirements.txt for this):
#   pip uninstall paddlepaddle -y
#   pip install paddlepaddle-gpu==3.0.0 -i https://www.paddlepaddle.org.cn/packages/stable/cu118/
#
# Step 3 — Install everything else from this file:
#   pip install -r requirements.txt
#
# Step 4 — Fix PaddlePaddle CUDA DLL loading on Windows (one-time only):
#   copy "venv\Lib\site-packages\nvidia\cudnn\bin\*.dll" "venv\Lib\site-packages\paddle\"
#   copy "venv\Lib\site-packages\nvidia\cublas\bin\*.dll" "venv\Lib\site-packages\paddle\"
#   copy "venv\Lib\site-packages\nvidia\cuda_runtime\bin\*.dll" "venv\Lib\site-packages\paddle\"
#
# Step 5 — Install Tesseract binary (separate from pip):
#   Download from: https://github.com/UB-Mannheim/tesseract/wiki
#   Add install folder to system PATH
#
# Step 6 — Verify GPU is detected:
#   python -c "import torch; print(torch.cuda.is_available())"        # must print True
#   python -c "import paddle; print(paddle.device.get_device())"      # must print gpu:0
#
